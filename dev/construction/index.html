from Integers, Floats and Irrationals

using DoubleFloats

a = Double64(pi)
b = Double32("5.12345") # prevent early conversion to Float64

a = DblF64"pi"

c = DblF16"1.125"

from two Reals

using DoubleFloats

a = 10.0
b = cbrt(10.0)
c = DoubleFloat(a, b)
